    function calculateShipping() {

      const miniBoxVol = 11.8 * 8.8 * 3.8;

      const squareBoxVol = 14.9 * 16.1 * 3.8;

      const coasterVol = 10 * 10 * 1.2;


      const miniQty = parseInt(document.getElementById("miniBox").value) || 0;

      const squareQty = parseInt(document.getElementById("squareBox").value) || 0;

      const coasterQty = parseInt(document.getElementById("coaster").value) || 0;


      const totalVol = miniBoxVol * miniQty + squareBoxVol * squareQty + coasterVol * coasterQty;

      const estimatedBoxVol = totalVol / 0.8;


      const cubeRoot = Math.cbrt(estimatedBoxVol);

      const width = cubeRoot;

      const length = cubeRoot * 1.5;

      const height = cubeRoot;

      const volumeWeight = (length * width * height) / 6000;


      let method = "";

      let cost = 0;


      const max711 = {

        weight: 5,

        size: 105,

        longest: 45

      };


      const totalSize = length + width + height;

      const longestSide = Math.max(length, width, height);


      if (volumeWeight <= max711.weight && totalSize <= max711.size && longestSide <= max711.longest) {

        method = "7-11 交貨便";

        cost = 60;

      } else {

        method = "黑貓宅急便";

        if (volumeWeight <= 5) cost = 170;

        else if (volumeWeight <= 10) cost = 210;

        else if (volumeWeight <= 15) cost = 250;

        else if (volumeWeight <= 20) cost = 290;

        else cost = 330;

      }


      cost += 10; // 加收 NT$10 手續費


      document.getElementById("result").innerHTML = `

        👉 預估包裝體積：約 ${estimatedBoxVol.toFixed(0)} cm³<br>

        👉 預估尺寸（長×寬×高）：${length.toFixed(1)} × ${width.toFixed(1)} × ${height.toFixed(1)} cm<br>

        📦 計費重：約 ${volumeWeight.toFixed(2)} kg<br>

        🚚 建議寄送方式：<strong>${method}</strong><br>

        💰 預估運費（含手續費）：<strong>NT$${cost}</strong>

      `;

    }
